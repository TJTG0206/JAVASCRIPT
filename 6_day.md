### 1. if - else

```javascript
  // 1. 일반적인 if-else 
  if(true){
    console.log("Hi!!");
  }else{
    console.log("HELLO!!");
  }
  
  // 2. && 와 ||
  if(true && true){
    console.log("Hi!!");
  }else{  
    console.log("HELLO!!");
  }
  
  /*
   * true && true = true;   
   * true && false = false;
   * false && true = false;
   * false && false = false;   
   */
   
   /*
   * true || true = true;   
   * true || false = true;
   * false || true = true;
   * false || false = false;   
   */
   
   prompt("오오오오오오!! 옛날꺼!");
```
